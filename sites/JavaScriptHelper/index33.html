Создание массива.

Приведем пример простого массива. При изменении номера массива в Mass1 будет соответственно меняться значение в Mass2

var Mass1, Mass2, Str, Mass3;
Mass1 = [0, 1, 2, 3, 4, 5, 6];
Mass2 = ["", "Март", "Апрель", "Май", "Июнь", "Июль", "Август"];
Mass3 = [6, 7, 8];
Mass1[0] = Mass3;
document.write(Mass1[0][2] + " месяц " + Mass2[1]);
			
var Mass1, Mass2, Str, Mass3;
Mass1 = [0, 11, 12, 13, 14, 15, 16];
Mass2 = ["", "Март", "Апрель", "Май", "Июнь", "Июль", "Август"];
Mass3 = [13, 14, 15];
Mass1[4] = Mass3;
document.write(Mass1[2][3] + " месяц " + Mass2[3]);
			
var Mass1, Mass2, Str, Mass3;
Mass1 = [0, 1, 2, 3, 4, 5];
Mass2 = ["", "Понедельник", "Вторник", "Среда", "Четверг", "Пятница"];
Mass3 = [3, 4, 5];
Mass1[4] = Mass3;
document.write(Mass1[1][1] + " месяц " + Mass2[1]);
