Усложнение массива.

Метод .join присваевает ссылку массива, метод .slice возвращает срез массива.

var Mass1, Mass2;
Mass1 = [[0, 1], 2, 3, 4];
Mass2 = Mass1.slice(0);
Mass2[0][0] = "Новое значение1 ";
Mass2[1] = "Новое значение2 "
document.write(Mass1.join(", ") + "br");
document.write(Mass2.join(", "));
			
var Mass1, Mass2;
Mass1 = [[0, 1], 2, 3, 4];
Mass2 = Mass1;
Mass2[0][0] = "Новое значение1 ";
Mass2[1] = "Новое значение2 "
document.write(Mass1.join(", ") + "br");
document.write(Mass2.join(", "));
