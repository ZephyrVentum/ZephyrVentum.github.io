Диалоговое окно Prompt.
Prompt является таким же диалоговым окном как Alert и Confirm, но в отличии от них он имеет текстовое поле. 

var n = window.prompt("Введите ваше имя.");
if (n==null) {
document.write("Вы нажали Cancel. ");
}
else {
document.write("Привет," + n + "!" + "br");
}
